def int_or_str(item):
    try:
        item = int(item)
    except:
        item = str(item)
    return item

def get_missing_value_info(num_vars, missing_values):
    
    missing_values = missing_values.copy()

    for n in range(num_vars):
        if 'X' in missing_values[n]:
            missing_values[n] = missing_values[n].replace('-1,', '').replace(' ','')

        str_values = missing_values[n].strip('][').split(',')
        missing_values[n] = list(map(int_or_str, str_values))

        if '' in missing_values[n]:
            missing_values[n] = [np.nan]
    return missing_values

def replace_missing_values(df, n_cols, num_vars, missing_values):
    
    df_copy = df.copy()
    get_missing_values = get_missing_value_info(num_vars, missing_values)

    for col_idx in range(n_cols):
        miss_value = get_missing_values[col_idx]
        column_name = df_copy.columns[col_idx]
        for value in miss_value:
            mask = df_copy[column_name] == value
            df_copy.loc[mask, column_name] = np.nan
    return df_copy

def engineer_vars(df,dicts, col_names, column, divide_by_ten=False):
    
    first_dict = dicts[0]
    second_dict = dicts[1]

    first_list = []
    second_list = []

    for value in df[column]:
        if np.isnan(float(value)):
            first_list.append(np.nan)
            second_list.append(np.nan)
        else:
            if divide_by_ten:
                value = int(value) // 10
                first_list.append(first_dict[int(value)])
                second_list.append(second_dict[int(value)])
            else:
                first_list.append(first_dict[int(value)])
                second_list.append(second_dict[int(value)])
                
    first_df = pd.DataFrame.from_dict({col_names[0]: first_list})
    second_df = pd.DataFrame.from_dict({col_names[1]: second_list})
    
    final_df = pd.concat([df, first_df, second_df], axis=1, sort=False)
    final_df = pd.get_dummies(final_df, columns=col_names)
    final_df = final_df.drop([column], axis=1)
    
    return final_df

Step 2.3: Interpret Principal Components¶

Now that we have our transformed principal components, it's a nice idea to check out the weight of each variable on the first few components to see if they can be interpreted in some fashion.

As a reminder, each principal component is a unit vector that points in the direction of highest variance (after accounting for the variance captured by earlier principal components). The further a weight is from zero, the more the principal component is in the direction of the corresponding feature. If two features have large weights of the same sign (both positive or both negative), then increases in one tend expect to be associated with increases in the other. To contrast, features with different signs can be expected to show a negative correlation: increases in one variable should result in a decrease in the other.
